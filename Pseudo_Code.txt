main c'est
début 
	resSwitch: entier;
	ent, binaire: entier;
	hexadec: chaine(MAX_HEXA);

	répéter
		affichMenu();
		lireClavier(resSwitch);
	jusqu'à (resSwitch>=1 ET resSwitch<=3)

	selon resSwitch faire
		cas 1:	ecrireEcran("Entrer le chiffre décimal :");
			lireClavier(ent);
			binaire := entToBin(entE decimal);
			hexadec := entToHexa(entE decimal);

		cas 2:	ecrireEcran("Entrer le chiffre binaire :");
			lireClavier(binaire);
			decimal := binToEnt(entE binaire);
			hexadec := binToHexa(ente binaire);

		cas 3:	ecrireEcran("Entrer le chiffre hexadécimal :");
			lireClavier(hexadec);
			decimal := hexaToEnt(entE hexadec);
			binaire := hexaToBin(entE hexadec);

		default : ecrireEcran("Erreur : Veuillez rentrer 1, 2 ou 3");
	finSelon

	affichRes(entE ent, entE binaire, entE hexadec);
fin
